# Function to perform ADF test and print results
def adf_test(series, name):
    result = adfuller(series, autolag='AIC')
    print(f'\nADF Test for {name}:')
    print(f'Test Statistic: {result[0]:.4f}')
    print(f'P-value: {result[1]:.4f}')
    print(f'Critical Values:')
    for key, value in result[4].items():
        print(f'   {key}: {value:.4f}')
    if result[1] < 0.05:
        print(f'Conclusion: {name} is stationary at 5% significance level (p-value < 0.05).')
    else:
        print(f'Conclusion: {name} is non-stationary at 5% significance level (p-value >= 0.05).')
    return result

# Perform ADF tests on GS3M and GS6M
adf_gs3m = adf_test(df['GS3M'], 'TB3M')
adf_gs6m = adf_test(df['GS6M'], 'TB6M')

# Cointegration test: Regress GS6M on GS3M
X = sm.add_constant(df['GS3M'])
model = OLS(df['GS6M'], X).fit()
residuals = model.resid

# Print OLS regression results
print('\nOLS Regression Results (GS6M on GS3M):')
print(model.summary())

# ADF test on residuals
print('\nCointegration Test (ADF on Residuals):')
adf_residuals = adf_test(residuals, 'Residuals')

# Economic interpretation of cointegration
print('\nEconomic Interpretation of Cointegration:')
print('Cointegration between TB3M and TB6M implies a long-run equilibrium relationship between the 3-month and 6-month Treasury bill rates. If the series are cointegrated, it suggests that while the two interest rates may deviate from each other in the short term due to market fluctuations or other factors, they tend to move together over the long term, maintaining a stable relationship. Economically, this reflects the interconnected nature of short-term and mid-term interest rates, as both are influenced by similar macroeconomic factors such as monetary policy, inflation expectations, and economic conditions. The presence of cointegration indicates that deviations from this equilibrium are temporary and corrected over time.')

# Model recommendation
print('\nModel Recommendation:')
if adf_residuals[1] < 0.05:
    print('Since the residuals of the OLS regression are stationary (p-value < 0.05), TB3M and TB6M are cointegrated.')
    print('Recommendation: A Vector Error Correction Model (VECM) is appropriate because it accounts for both the long-run cointegration relationship and short-run dynamics between the series.')
elif adf_gs3m[1] >= 0.05 or adf_gs6m[1] >= 0.05:
    print('Since one or both series are non-stationary (p-value >= 0.05) and not cointegrated, a VAR model in first differences is recommended to ensure stationarity.')
else:
    print('Since both series are stationary (p-value < 0.05), a VAR model in levels is appropriate.')
